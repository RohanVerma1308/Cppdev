Hybrid Inheritance

Shows hybrid inheritance which is a combination of multiple types of inheritance.

#include <iostream>
using namespace std;

// Base class
class Animal {
public:
    void breathe() {
        cout << "Animal breathes" << endl;
    }
};

// Derived class 1 from Animal
class Mammal : public Animal {
public:
    void feedMilk() {
        cout << "Mammal feeds milk to young ones" << endl;
    }
};

// Derived class 2 from Animal
class Carnivore : public Animal {
public:
    void eatMeat() {
        cout << "Carnivore eats meat" << endl;
    }
};

// Child class inherits from both Mammal and Carnivore (multiple inheritance)
class Tiger : public Mammal, public Carnivore {
public:
    void roar() {
        cout << "Tiger roars" << endl;
    }
};

int main() {
    Tiger t;

    // From Animal via Mammal
    t.Mammal::breathe();  // Need to specify which path to use to resolve ambiguity
    
    // From Mammal
    t.feedMilk();
    
    // From Carnivore
    t.eatMeat();
    
    // Tiger's own method
    t.roar();

    return 0;
}

Overloading Of A Binary Arithmetic Operator

Shows how to overload binary operators like +, -, *, etc. in C++.

#include <iostream>
using namespace std;

class Complex {
private:
    double real;
    double imag;
    
public:
    // Constructor
    Complex(double r = 0.0, double i = 0.0) : real(r), imag(i) {}
    
    // Display function
    void display() {
        cout << real << " + " << imag << "i" << endl;
    }
    
    // Overload + operator
    Complex operator+(const Complex &obj) {
        Complex temp;
        temp.real = real + obj.real;
        temp.imag = imag + obj.imag;
        return temp;
    }
};

int main() {
    Complex c1(2.5, 3.5), c2(1.5, 2.5), c3;
    
    cout << "First complex number: ";
    c1.display();
    
    cout << "Second complex number: ";
    c2.display();
    
    // Using overloaded + operator
    c3 = c1 + c2;
    
    cout << "Result of addition: ";
    c3.display();
    
    return 0;
}

Overloading Of A Unary Arithmetic Operator

Demonstrates how to overload unary operators like ++, --, etc. in C++.

#include <iostream>
using namespace std;

class Counter {
private:
    int count;
    
public:
    // Constructor
    Counter(int c = 0) : count(c) {}
    
    // Display function
    void display() {
        cout << "Count: " << count << endl;
    }
    
    // Overload prefix ++ operator
    Counter operator++() {
        ++count;
        return *this;
    }
    
    // Overload postfix ++ operator
    Counter operator++(int) {
        Counter temp = *this;
        count++;
        return temp;
    }
    
    // Overload unary - operator
    Counter operator-() {
        Counter temp;
        temp.count = -count;
        return temp;
    }
};

int main() {
    Counter c1(5), c2, c3;
    
    cout << "Initial values:" << endl;
    cout << "c1 ";
    c1.display();
    
    // Using prefix ++ operator
    c2 = ++c1;
    cout << "\nAfter c2 = ++c1:" << endl;
    cout << "c1 ";
    c1.display();
    cout << "c2 ";
    c2.display();
    
    // Using postfix ++ operator
    c3 = c1++;
    cout << "\nAfter c3 = c1++:" << endl;
    cout << "c1 ";
    c1.display();
    cout << "c3 ";
    c3.display();
    
    // Using unary - operator
    c2 = -c1;
    cout << "\nAfter c2 = -c1:" << endl;
    cout << "c1 ";
    c1.display();
    cout << "c2 ";
    c2.display();
    
    return 0;
}

Read Contents From A Specific File

Shows how to read data from a file in C++.

#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    string filename;
    cout << "Enter the filename to read: ";
    getline(cin, filename);
    
    // Open the file
    ifstream file(filename);
    
    // Check if file opened successfully
    if (!file.is_open()) {
        cerr << "Error opening file: " << filename << endl;
        return 1;
    }
    
    cout << "\nFile contents:" << endl;
    cout << "-------------" << endl;
    
    string line;
    int lineCount = 0;
    
    // Read file line by line
    while (getline(file, line)) {
        lineCount++;
        cout << lineCount << ": " << line << endl;
    }
    
    // Close the file
    file.close();
    
    cout << "-------------" << endl;
    cout << "Total lines: " << lineCount << endl;
    
    return 0;
}

Virtual Base Class

Demonstrates the concept of virtual base classes to solve diamond problem in multiple inheritance.

#include <iostream>
using namespace std;

// Base class
class Animal {
public:
    Animal() {
        cout << "Animal constructor called" << endl;
    }
    
    void eat() {
        cout << "Animal eats food" << endl;
    }
};

// First derived class - using virtual inheritance
class Mammal : virtual public Animal {
public:
    Mammal() {
        cout << "Mammal constructor called" << endl;
    }
};

// Second derived class - using virtual inheritance
class Bird : virtual public Animal {
public:
    Bird() {
        cout << "Bird constructor called" << endl;
    }
};

// Class inheriting from both Mammal and Bird
class Bat : public Mammal, public Bird {
public:
    Bat() {
        cout << "Bat constructor called" << endl;
    }
};

int main() {
    cout << "Creating Bat object:" << endl;
    Bat b;
    
    // Without virtual inheritance, this would be ambiguous
    b.eat();
    
    return 0;
}

Write Data To A File

Shows how to write data to a file in C++.

#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    string filename;
    
    cout << "Enter the filename to write to: ";
    getline(cin, filename);
    
    // Open file in output mode
    ofstream outFile(filename);
    
    // Check if file opened successfully
    if (!outFile.is_open()) {
        cerr << "Error opening file: " << filename << endl;
        return 1;
    }
    
    string input;
    cout << "\nEnter text to write to the file (type 'END' on a new line to stop):" << endl;
    
    while (true) {
        getline(cin, input);
        
        // Check for exit condition
        if (input == "END") {
            break;
        }
        
        // Write to file
        outFile << input << endl;
    }
    
    // Close the file
    outFile.close();
    
    cout << "\nData has been written to " << filename << endl;
    
    return 0;
}
